

cd-rpc cluster receives request from cd-cli or ai agent in the form of ICdRequest.
One of the core interfaces for ICdRequest payload is managed via IQuery interface.
cd-rpc is then able to process cd-bio-engine requests. 
Notes that cd-rpc is stationed at the infrastrucure edge and handles consumer resources.
Corpdesk centralized services are however executed by cd-api cluster.

Now that IQuery has been updated, cd-rpc needs to comply with the new changes. 
The updates on IQuery focussed on json handling based on SNP protocol.


cd-rpc needs to be upgraded to to process and post request in the new format.
Other tasks that need to be carried is adding validation to SNP processing.

Upgraded IQuery interface:
```ts
export interface IQuery {
    select?: string[];

    update?: Record<string, any>;

    where?: IQueryWhere;

    snpWhere?: SnpSelector[];

    snpOrWhere?: SnpSelector[];

    snpCrud?: SnpInstruction[];

    distinct?: boolean;

    take?: number;

    skip?: number;

    jFilters?: IJFilter[];

    order?: any;

    class?: string;

    extData?: any;
}
```

```ts
// src/CdNode/sys/base/models/jsdp.model.ts

/**
 * SNP Interfaces
 */

export interface SnpSelector {
  modelField?: string;
  path: string[];
  value: any;
  op?: "eq" | "contains";
}

export type SnpPathSegment = string | number | SnpSelector;

export interface SnpFilter {
  v?: string;
  modelField?: string; // the field in the datasource that contains the semantic data eg typeorm entyty field that has the json data
  path: SnpPathSegment[];
  value?: any;
  op: "eq" | "neq" | "gt" | "gte" | "lt" | "lte" | "contains" | "exists";
}

export interface SnpInstruction {
  v?: string;
  modelField?: string;
  path: SnpPathSegment[];
  value?: any;
  action: "create" | "read" | "update" | "delete" | "upsert";
  conditions?: any;
}


export interface ISnpRootTarget {
  isRoot: true;
  value: any;
}

export interface ISnpChildTarget {
  isRoot: false;
  parent: any;
  key: string | number;
  value: any;
}

export type ISnpResolvedTarget = ISnpRootTarget | ISnpChildTarget;

export interface ISnpExecutionResult {
  success: boolean;
  matchesWhere: boolean;
  matchesOrWhere: boolean;
  modified: boolean;
  reads: any[];
  errors: string[];
}


export interface ISnpDataSource {
  find(selectors: SnpSelector[]): Promise<any[]>;

  save(items: any[]): Promise<void>;
}

export interface ISnpFormatAdapter {
  getValueByPath(root: any, path: string[]): any;

  setValueByPath(root: any, path: string[], value: any): void;

  deleteValueByPath(root: any, path: string[]): void;

  toNative(data: any): any;

  fromNative(data: any): any;
}
```
